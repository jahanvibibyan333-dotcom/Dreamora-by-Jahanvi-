/* Expandable product items */
function toggleItems(btn){
    const itemList = btn.nextElementSibling;
    if(itemList.style.display === "block"){
        itemList.style.display = "none";
        btn.textContent = "Show Items";
    } else {
        itemList.style.display = "block";
        btn.textContent = "Hide Items";
    }
}

/* Review slider */
let currentSlide = 0;
const slides = document.querySelector('.slides');
function showSlide(index){
    const total = slides.children.length;
    if(index >= total) currentSlide = 0;
    else if(index < 0) currentSlide = total-1;
    else currentSlide = index;
    slides.style.transform = `translateX(-${currentSlide*100}%)`;
}
function nextSlide(){ showSlide(currentSlide+1); }
function prevSlide(){ showSlide(currentSlide-1); }

/* Simple cart */
const cartDiv = document.getElementById('cart');
const cartButtons = document.querySelectorAll('.cart-btn');

cartButtons.forEach(btn => {
    btn.addEventListener('click', () => {
        const productName = btn.parentElement.querySelector('h3').textContent;
        const packaging = btn.parentElement.querySelector('select').value;
        const item = document.createElement('p');
        item.textContent = `${productName} | Packaging: ${packaging}`;
        cartDiv.appendChild(item);
    });
});
