/* Cart logic */
let CART = [];
const cartDrawer = document.getElementById('cartDrawer');
const cartList = document.getElementById('cartList');
const cartCount = document.getElementById('cartCount');
const subtotalEl = document.getElementById('subtotal');

function findProduct(id){ return PRODUCTS.find(x=>x.id===id) }

function updateCartUI(){
  cartList.innerHTML='';
  let subtotal = 0;
  CART.forEach((item, idx)=>{
    const row = document.createElement('div');
    row.className='cart-item';
    row.innerHTML = `
      <img src="" alt="${item.title}">
      <div style="flex:1">
        <div style="display:flex;justify-content:space-between;align-items:center">
          <div><strong>${item.title}</strong><div class="small">${item.variant}</div></div>
          <div>₹${item.price*item.qty}</div>
        </div>
        <div style="display:flex;gap:8px;margin-top:6px;align-items:center">
          <input type="number" min="1" value="${item.qty}" style="width:64px;padding:6px;border-radius:8px;border:1px solid #eee" data-idx="${idx}" class="cartQty">
          <button data-idx="${idx}" class="removeBtn" style="padding:6px;border-radius:8px">Remove</button>
        </div>
      </div>
    `;
    cartList.appendChild(row);
    subtotal += item.price * item.qty;
  });
  // packaging
  const pack = document.querySelector('input[name="pack"]:checked')?.value || 'none';
  let packPrice = 0;
  if(pack==='box') packPrice = 35;
  if(pack==='bag') packPrice = 25;
  subtotalEl.textContent = `₹${subtotal + packPrice}`;
  cartCount.textContent = CART.reduce((s,i)=>s+i.qty,0);
  if(CART.length===0) cartList.innerHTML = '<div class="small">Cart empty. Add some hampers!</div>';
}

/* Events */
document.body.addEventListener('click', function(e){
  if(e.target.classList.contains('addToCartBtn')){
    const pid = e.target.getAttribute('data-product');
    const variant = document.querySelector(`select[data-product="${pid}"]`).value;
    const qty = parseInt(document.querySelector(`input.qtyInput[data-product="${pid}"]`).value) || 1;
    const prod = findProduct(pid);
    if(pid==='return1' && qty < 5){
      alert('Return Gift 1 minimum order quantity is 5.');
      return;
    }
    const existing = CART.find(i=>i.id===pid && i.variant===variant);
    if(existing){
      existing.qty += qty;
    } else {
      CART.push({ id: pid, title: prod.title, price: prod.price, variant, qty });
    }
    updateCartUI();
    alert(`${prod.title} added to cart`);
  }

  if(e.target.id==='openCart'){
    cartDrawer.style.display='flex';
    cartDrawer.setAttribute('aria-hidden','false');
    updateCartUI();
  }
  if(e.target.id==='closeCart'){
    cartDrawer.style.display='none';
    cartDrawer.setAttribute('aria-hidden','true');
  }
  if(e.target.classList.contains('removeBtn')){
    const idx = parseInt(e.target.getAttribute('data-idx'));
    CART.splice(idx,1);
    updateCartUI();
  }
});

cartList.addEventListener('change', function(e){
  if(e.target.classList.contains('cartQty')){
    const idx = parseInt(e.target.getAttribute('data-idx'));
    let v = parseInt(e.target.value);
    if(v < 1) v = 1;
    if(CART[idx].id === 'return1' && v < 5){
      alert('Return Gift 1 minimum order quantity is 5.');
      e.target.value = 5;
      v = 5;
    }
    CART[idx].qty = v;
    updateCartUI();
  }
});

document.querySelectorAll('input[name="pack"]').forEach(el=>{
  el.addEventListener('change', updateCartUI);
});

const reviews = ['review0','review1','review2'];
function showReview(i){ reviews.forEach((id,idx)=>{ document.getElementById(id).style.display = idx===i ? 'block' : 'none'; }); }
document.getElementById('revPrev').addEventListener('click', ()=>{ currentReview = (currentReview-1+reviews.length)%reviews.length; showReview(currentReview);});
document.getElementById('revNext').addEventListener('click', ()=>{ currentReview = (currentReview+1)%reviews.length; showReview(currentReview);});
setInterval(()=>{ currentReview = (currentReview+1)%reviews.length; showReview(currentReview); }, 5000);

/* initial */
updateCartUI(); showReview(0);
</script>
